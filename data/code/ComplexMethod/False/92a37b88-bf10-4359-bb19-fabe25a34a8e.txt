public class AnnotationReturnModel {

	private int size;
	
	private Sample sample;

	/**
	 * @return the size
	 */
	public int getSize() {
		return size;
	}

	/**
	 * @param size the size to set
	 */
	public void setSize(int size) {
		this.size = size;
	}

	/**
	 * @return the sample
	 */
	public Sample getSample() {
		return sample;
	}

	/**
	 * @param sample the sample to set
	 */
	public void setSample(Sample sample) {
		this.sample = sample;
	}
	
}
